## 四、BaseField 基类设计

```python
class BaseField:
    """字段类型基类"""
    name = None           # 机器名，如 string, integer
    label = None         # 显示标签，如 "单行文本"
    widget = None        # 前端组件，如 input, select
    properties = []      # 内部属性列表
    
    def __init__(self, field_name, field_config):
        self.field_name = field_name
        self.field_config = field_config  # JSON 配置
    
    def render(self, value, mode='edit'):
        """渲染 HTML
        - value: 当前值（dict）
        - mode: edit/view
        """
        raise NotImplementedError
    
    def validate(self, value):
        """验证数据，返回错误列表"""
        return []
    
    def format(self, value):
        """格式化显示"""
        return value
    
    def get_widget_config(self):
        """获取前端组件配置"""
        return {}
```

## 六、字段实现示例

### 6.1 String 字段

```python
# app/modules/core/fields/string.py
from .base import BaseField

class StringField(BaseField):
    name = 'string'
    label = '单行文本'
    widget = 'input'
    properties = ['value']
    
    def render(self, value, mode='edit'):
        if mode == 'view':
            return value.get('value', '')
        
        return f'<input type="text" name="{self.field_name}" value="{value.get("value", "")}" class="form-control">'
    
    def validate(self, value):
        errors = []
        if self.field_config.get('required') and not value.get('value'):
            errors.append(f'{self.field_config.get("label")} 为必填项')
        return errors
```

### 6.2 Email 字段

```python
# app/modules/core/fields/email.py
from .base import BaseField

class EmailField(BaseField):
    name = 'email'
    label = '邮箱'
    widget = 'input'
    properties = ['value']
    
    def render(self, value, mode='edit'):
        if mode == 'view':
            return value.get('value', '')
        
        return f'<input type="email" name="{self.field_name}" value="{value.get("value", "")}" class="form-control">'
    
    def validate(self, value):
        import re
        errors = []
        email = value.get('value', '')
        if email and not re.match(r'^[\w\.-]+@[\w\.-]+\.\w+$', email):
            errors.append(f'{self.field_config.get("label")} 格式不正确')
        return errors
```

### 6.3 Address 字段

```python
# app/modules/core/fields/address.py
from .base import BaseField

class AddressField(BaseField):
    name = 'address'
    label = '标准地址'
    widget = 'address_input'
    properties = ['province', 'city', 'district', 'street', 'house_number', 'grid_id']
    
    def render(self, value, mode='edit'):
        if mode == 'view':
            return f"{value.get('province', '')}{value.get('city', '')}{value.get('district', '')}{value.get('street', '')}{value.get('house_number', '')}"
        
        # 渲染地址选择器
        return f'''
        <div class="address-field">
            <input type="text" name="{self.field_name}[province]" placeholder="省" value="{value.get('province', '')}">
            <input type="text" name="{self.field_name}[city]" placeholder="市" value="{value.get('city', '')}">
            <input type="text" name="{self.field_name}[district]" placeholder="区" value="{value.get('district', '')}">
            <input type="text" name="{self.field_name}[street]" placeholder="街道" value="{value.get('street', '')}">
            <input type="text" name="{self.field_name}[house_number]" placeholder="门牌号" value="{value.get('house_number', '')}">
        </div>
        '''
```

## 七、模块注册器

```python
# app/modules/core/fields/__init__.py
from .base import BaseField
from .string import StringField
from .email import EmailField
# ... 导入所有字段

FIELD_TYPES = {
    'string': StringField,
    'email': EmailField,
    # ... 注册所有字段
}

def get_field_type(name):
    return FIELD_TYPES.get(name, StringField)

def get_all_field_types():
    return FIELD_TYPES
```
