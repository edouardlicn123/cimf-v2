## 一、功能概述

数据导入功能允许用户上传 CSV 或 XLSX 格式的文件，将数据批量导入到已启用的节点模块中。

### 1.1 核心特点

- **通用性**：适用于所有已启用的节点模块
- **文件支持**：支持 CSV 和 XLSX 格式
- **字段映射**：文件列可映射到目标字段
- **数据预览**：导入前预览数据并验证
- **冲突处理**：唯一字段冲突时更新现有记录
- **错误报告**：详细的导入结果和错误列表

### 1.2 适用场景

- 从其他系统迁移数据
- 批量录入新数据
- 批量更新现有数据

```python
        ('success', '成功'),

        NodeType,

        null=True,
        verbose_name='节点类型'
    )

    status = models.CharField(
        max_length=20, 
        choices=STATUS_CHOICES, 
        default='pending',
        verbose_name='状态'
    )
    total_rows = models.IntegerField(default=0, verbose_name='总行数')
    success_count = models.IntegerField(default=0, verbose_name='成功数')
    error_count = models.IntegerField(default=0, verbose_name='错误数')
    error_details = models.JSONField(default=list, verbose_name='错误详情')
    imported_by = models.ForeignKey(
        settings.AUTH_USER_MODEL,
        on_delete=models.SET_NULL,
        null=True,
        verbose_name='导入人'
    )
    created_at = models.DateTimeField(auto_now_add=True, verbose_name='创建时间')
    completed_at = models.DateTimeField(null=True, blank=True, verbose_name='完成时间')
    
    class Meta:
        ordering = ['-created_at']
        verbose_name = '导入日志'
        verbose_name_plural = '导入日志'
```

```python
# validators.py

class FieldValidator:
    """字段验证器"""
    
    @staticmethod
    def string(value: str, max_length: int = None) -> List[str]:
        errors = []
        if not isinstance(value, str):
            errors.append('必须为字符串')
        if max_length and len(value) > max_length:
            errors.append(f'长度不能超过 {max_length}')
        return errors
    
    @staticmethod
    def email(value: str) -> List[str]:
        errors = []
        if value and not re.match(r'^[\w\.-]+@[\w\.-]+\.\w+$', value):
            errors.append('邮箱格式不正确')
        return errors
    
    @staticmethod
    def telephone(value: str) -> List[str]:
        errors = []
        if value and not re.match(r'^[\d\-\+\(\)\s]+$', value):
            errors.append('电话号码格式不正确')
        return errors
    
    @staticmethod
    def fk_lookup(slug: str, value: str):
        """验证 FK 字段，尝试通过名称查找"""
        return TaxonomyService.get_item_by_name(slug, value)
```

### 6.1 视图函数

```python
# nodes/views.py

@login_required
def import_list(request):
    """导入页 - 显示所有模块的导入入口"""
    node_types = NodeTypeService.get_all()
    return render(request, 'core/structure/importexport/import.html', {
        'node_types': node_types,
        'active_section': 'import',
    })

@login_required
def import_page(request, node_type_slug):
    """导入操作页"""
    node_type = NodeTypeService.get_by_slug(node_type_slug)
    if not node_type:
        return redirect('nodes:import_list')
    
    # 获取可导入的字段
    fields = ImportService.get_importable_fields(node_type_slug)
    
    return render(request, 'core/structure/importexport/import_do.html', {
        'node_type': node_type,
        'fields': fields,
    })

@login_required
def preview_import(request, node_type_slug):
    """预览导入数据 - AJAX"""
    if request.method != 'POST':
        return JsonResponse({'error': 'POST required'}, status=400)
    
    node_type = NodeTypeService.get_by_slug(node_type_slug)
    if not node_type:
        return JsonResponse({'error': 'Node type not found'}, status=404)
    
    file = request.FILES.get('file')
    if not file:
        return JsonResponse({'error': 'No file uploaded'}, status=400)
    
    mapping = request.POST.get('mapping', '[]')
    mapping = json.loads(mapping)
    
    skip_header = request.POST.get('skip_header', 'true') == 'true'
    
    result = ImportService.preview_data(
        file, 
        file.name.split('.')[-1],
        node_type_slug,
        mapping,
        skip_header
    )
    
    return JsonResponse(result)

@login_required
def do_import(request, node_type_slug):
    """执行导入"""
    if request.method != 'POST':
        return redirect('nodes:import_page', node_type_slug)
    
    node_type = NodeTypeService.get_by_slug(node_type_slug)
    if not node_type:
        return redirect('nodes:import_list')
    
    file = request.FILES.get('file')
    if not file:
        messages.error(request, '请上传文件')
        return redirect('nodes:import_page', node_type_slug)
    
    mapping = request.POST.get('mapping', '[]')
    mapping = json.loads(mapping)
    
    skip_header = request.POST.get('skip_header', 'true') == 'true'
    update_existing = request.POST.get('update_existing', 'false') == 'true'
    
    import_log = ImportService.import_data(
        file,
        file.name.split('.')[-1],
        node_type_slug,
        mapping,
        request.user,
        skip_header,
        update_existing
    )
    
    messages.success(
        request, 
        f'导入完成：成功 {import_log.success_count} 条，失败 {import_log.error_count} 条'
    )
    
    return render(request, 'core/structure/importexport/import_result.html', {
        'import_log': import_log,
    })
```

## 八、关键技术点

### 8.1 唯一字段冲突处理

```python
# 导入时查找现有记录
unique_fields = ['customer_code']  # 唯一字段列表

for row in data:
    # 尝试查找现有记录
    existing = CustomerFields.objects.filter(
        customer_code=row['customer_code']
    ).first()
    
    if existing:
        if update_existing:
            # 更新现有记录
            for field, value in row.items():
                setattr(existing, field, value)
            existing.save()
            updated_count += 1
        else:
            # 跳过
            skipped_count += 1
    else:
        # 创建新记录
        CustomerFields.objects.create(**row)
        created_count += 1
```
