### 3.1 SettingsService

```python
# core/services/settings_service.py

from core.models import SystemSetting
from django.core.cache import cache

class SettingsService:
    DEFAULT_SETTINGS = {
        'system_name': 'CIMF',
        'upload_max_size_mb': '12',
        # ... 其他默认设置
    }
    
    CACHE_KEY = 'system_settings'
    CACHE_TTL = 60
    
    @classmethod
    def get_setting(cls, key, default=None):
        # 先从缓存获取
        settings = cls.get_all_settings()
        return settings.get(key, default)
    
    @classmethod
    def get_all_settings(cls):
        # 从缓存或数据库获取
        settings = cache.get(cls.CACHE_KEY)
        if settings is None:
            settings = {s.key: s.value for s in SystemSetting.objects.all()}
            # 合并默认值
            settings = {**cls.DEFAULT_SETTINGS, **settings}
            cache.set(cls.CACHE_KEY, settings, cls.CACHE_TTL)
        return settings
    
    @classmethod
    def save_setting(cls, key, value):
        obj, _ = SystemSetting.objects.update_or_create(
            key=key, defaults={'value': str(value)}
        )
        cache.delete(cls.CACHE_KEY)
        return obj
    
    @classmethod
    def clear_cache(cls):
        cache.delete(cls.CACHE_KEY)
```

### 3.2 PermissionService

```python
# core/services/permission_service.py

class PermissionService:
    SYSTEM_PERMISSIONS = [
        'system.view', 'system.manage',
        'permission.view', 'permission.manage',
        'user.create', 'user.view', 'user.edit', 'user.delete',
    ]
    
    @classmethod
    def get_user_permissions(cls, user):
        if user.is_superuser:
            return cls.SYSTEM_PERMISSIONS
        return user.permissions or []
    
    @classmethod
    def has_permission(cls, user, permission):
        return permission in cls.get_user_permissions(user)
```

### 3.3 UserService

```python
# core/services/user_service.py

from django.contrib.auth import get_user_model
from django.contrib.auth.hashers import make_password

class UserService:
    @classmethod
    def create_user(cls, username, password, **kwargs):
        User = get_user_model()
        return User.objects.create_user(username, password, **kwargs)
    
    @classmethod
    def get_users(cls, **filters):
        return get_user_model().objects.filter(**filters)
```
