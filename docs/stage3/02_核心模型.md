### 2.1 User 模型（扩展 AbstractUser）

```python
# core/models.py

from django.contrib.auth.models import AbstractUser
from django.db import models

class User(AbstractUser):
    """扩展 Django 内置用户模型"""
    
    nickname = models.CharField(max_length=64, blank=True)
    role = models.CharField(max_length=20, default='employee')
    permissions = models.JSONField(default=list)
    failed_login_attempts = models.IntegerField(default=0)
    locked_until = models.DateTimeField(null=True, blank=True)
    theme = models.CharField(max_length=20, default='default')
    notifications_enabled = models.BooleanField(default=True)
    preferred_language = models.CharField(max_length=10, default='zh')
    last_login_at = models.DateTimeField(null=True, blank=True)
    updated_at = models.DateTimeField(auto_now=True)
```

### 2.2 SystemSetting 模型

```python
class SystemSetting(models.Model):
    """系统设置"""
    key = models.CharField(max_length=128, unique=True)
    value = models.TextField()
    description = models.CharField(max_length=255, blank=True)
    updated_at = models.DateTimeField(auto_now=True)
```

### 2.3 Taxonomy 词汇表

```python
class Taxonomy(models.Model):
    """词汇表"""
    name = models.CharField(max_length=128)
    slug = models.CharField(max_length=128, unique=True)
    description = models.CharField(max_length=512, blank=True)
    created_at = models.DateTimeField(auto_now_add=True)
    updated_at = models.DateTimeField(auto_now=True)

class TaxonomyItem(models.Model):
    """词汇项"""
    taxonomy = models.ForeignKey(Taxonomy, on_delete=models.CASCADE, related_name='items')
    name = models.CharField(max_length=256)
    description = models.CharField(max_length=512, blank=True)
    weight = models.IntegerField(default=0)
    created_at = models.DateTimeField(auto_now_add=True)
    updated_at = models.DateTimeField(auto_now=True)
```

### 2.5 配置 Django Admin

```python
# core/admin.py
from django.contrib import admin
from .models import User, SystemSetting, Taxonomy, TaxonomyItem

@admin.register(User)
class UserAdmin(admin.ModelAdmin):
    list_display = ['username', 'nickname', 'email', 'role', 'is_active']
    list_filter = ['role', 'is_active']
    search_fields = ['username', 'nickname', 'email']

@admin.register(SystemSetting)
class SystemSettingAdmin(admin.ModelAdmin):
    list_display = ['key', 'value', 'updated_at']
    search_fields = ['key']

@admin.register(Taxonomy)
class TaxonomyAdmin(admin.ModelAdmin):
    list_display = ['name', 'slug', 'created_at']
    prepopulated_fields = {'slug': ('name',)}

admin.site.register(TaxonomyItem)
```
