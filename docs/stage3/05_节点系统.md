```python
# nodes/models.py

class NodeType(models.Model):
    """节点类型"""
    name = models.CharField(max_length=100)
    slug = models.CharField(max_length=50, unique=True)
    description = models.CharField(max_length=500, blank=True)
    icon = models.CharField(max_length=50, default='bi-folder')
    fields_config = models.JSONField(default=dict)
    is_active = models.BooleanField(default=True)
    created_at = models.DateTimeField(auto_now_add=True)
    updated_at = models.DateTimeField(auto_now=True)
```

### 5.2 Node 基类

```python
class Node(models.Model):
    """节点主表"""
    node_type = models.ForeignKey(NodeType, on_delete=models.CASCADE)
    created_by = models.ForeignKey('core.User', on_delete=models.SET_NULL, null=True)
    updated_by = models.ForeignKey('core.User', on_delete=models.SET_NULL, null=True)
    created_at = models.DateTimeField(auto_now_add=True)
    updated_at = models.DateTimeField(auto_now=True)
    
    class Meta:
        abstract = True
```

```python
class Customer(Node):

    phone = models.CharField(max_length=20, blank=True)
    email = models.EmailField(blank=True)
    address = models.TextField(blank=True)
    customer_type = models.CharField(max_length=50, blank=True)
    notes = models.TextField(blank=True)
```
