# Node 节点系统实现指南

本文档以"客户信息"节点类型为例，展示完整的 Node/Taxonomy 系统实现流程。

---

## 一、概述

Node 系统是一个灵活的节点类型管理框架，允许通过管理界面创建各种业务节点类型，每个节点类型可以配置不同的字段。

- **节点类型机器名**：使用小写字母和下划线，如 `customer`、`project`、`order`

#### 1.3.2 节点类型管理服务

```python
# app/services/node_type_manager.py
from app.models.node import NodeType, Node
from app import db

class NodeTypeManager:
    """节点类型管理器 - 提供启用、禁用、删除功能"""
    
    @staticmethod
    def get_all_configs():
        """获取所有已注册的节点类型配置"""
        from importlib import import_module
        import os
        
        configs = []
        node_module_dir = 'app/modules/node'
        
        # 遍历 node 目录下的所有子模块
        for item in os.listdir(node_module_dir):
            item_path = os.path.join(node_module_dir, item)
            if os.path.isdir(item_path) and not item.startswith('_'):
                try:
                    module = import_module(f'app.modules.node.{item}')
                    if hasattr(module, 'NODE_TYPE_CONFIG'):
                        configs.append(module.NODE_TYPE_CONFIG)
                except ImportError:
                    continue
        
        return configs
    
    @staticmethod
    def enable(node_type_slug):
        """启用节点类型"""
        node_type = NodeType.query.filter_by(slug=node_type_slug).first()
        if node_type:
            node_type.is_active = True
            db.session.commit()
            return True
        return False
    
    @staticmethod
    def disable(node_type_slug):
        """禁用节点类型 - 禁用后该类型下的节点将不可访问"""
        node_type = NodeType.query.filter_by(slug=node_type_slug).first()
        if node_type:
            node_type.is_active = False
            db.session.commit()
            return True
        return False
    
    @staticmethod
    def delete(node_type_slug):
        """删除节点类型 - 删除前需确认无关联数据"""
        node_type = NodeType.query.filter_by(slug=node_type_slug).first()
        if not node_type:
            return False
        
        # 检查是否有关联的节点
        node_count = Node.query.filter_by(node_type_id=node_type.id).count()
        if node_count > 0:
            raise ValueError(f'该节点类型下仍有 {node_count} 条数据，无法删除')
        
        # 获取字段模型并删除字段表
        config = NodeTypeManager.get_config_by_slug(node_type_slug)
        if config:
            fields_table = config.get('tables', {}).get('fields')
            if fields_table:
                # 执行删除字段表的操作
                db.session.execute(db.text(f'DROP TABLE IF EXISTS {fields_table}'))
        
        # 删除节点类型记录
        db.session.delete(node_type)
        db.session.commit()
        return True
    
    @staticmethod
    def get_config_by_slug(slug):
        """根据 slug 获取节点类型配置"""
        from importlib import import_module
        try:
            module = import_module(f'app.modules.node.{slug}')
            if hasattr(module, 'NODE_TYPE_CONFIG'):
                return module.NODE_TYPE_CONFIG
        except ImportError:
            return None
        return None
    
    @staticmethod
    def is_enabled(node_type_slug):
        """检查节点类型是否启用"""
        node_type = NodeType.query.filter_by(slug=node_type_slug).first()
        return node_type and node_type.is_active
```

class NodeTypeService:
    """节点类型服务"""
    
    @staticmethod
    def get_all():
        return NodeType.query.filter_by(is_active=True).all()
    
    @staticmethod
    def get_by_id(node_type_id):
        return NodeType.query.get(node_type_id)
    
    @staticmethod
    def get_by_slug(slug):
        return NodeType.query.filter_by(slug=slug, is_active=True).first()
    
    @staticmethod
    def create(data):
        node_type = NodeType(**data)
        db.session.add(node_type)
        db.session.commit()
        return node_type
    
    @staticmethod
    def update(node_type_id, data):
        node_type = NodeType.query.get(node_type_id)
        for key, value in data.items():
            setattr(node_type, key, value)
        db.session.commit()
        return node_type
    
    @staticmethod
    def delete(node_type_id):
        node_type = NodeType.query.get(node_type_id)
        node_type.is_active = False
        db.session.commit()
```

#### 2.2.2 节点数据服务 `node_service.py`

```python
class NodeService:
    """节点数据服务 - 处理节点主表操作"""
    
    # 节点类型 slug 到字段模型的映射
    FIELDS_MODELS = {
        'customer': 'CustomerFields',
        'project': 'ProjectFields',
        'order': 'OrderFields',
    }
    
    @staticmethod
    def get_by_type(node_type_slug, page=1, per_page=20):
        node_type = NodeTypeService.get_by_slug(node_type_slug)
        if not node_type:
            return []
        return Node.query.filter_by(node_type_id=node_type.id).paginate(page=page, per_page=per_page)
    
    @staticmethod
    def get_by_id(node_id):
        return Node.query.get(node_id)
    
    @staticmethod
    def get_fields(node_id, node_type_slug):
        """获取节点的字段数据"""
        model_name = NodeService.FIELDS_MODELS.get(node_type_slug)
        if not model_name:
            return None
        fields_model = globals().get(model_name)
        if not fields_model:
            return None
        return fields_model.query.filter_by(node_id=node_id).first()
    
    @staticmethod
    def create(node_type_id, fields_data, user_id):
        """创建节点：先创建主表记录，再创建字段表记录"""
        node = Node(
            node_type_id=node_type_id,
            created_by=user_id,
            updated_by=user_id
        )
        db.session.add(node)
        db.session.flush()  # 获取 node.id
        
        # 根据节点类型创建字段记录
        node_type = NodeTypeService.get_by_id(node_type_id)
        model_name = NodeService.FIELDS_MODELS.get(node_type.slug)
        if model_name:
            fields_model = globals().get(model_name)
            if fields_model:
                fields_record = fields_model(node_id=node.id, **fields_data)
                db.session.add(fields_record)
        
        db.session.commit()
        return node
    
    @staticmethod
    def update(node_id, fields_data, user_id):
        """更新节点字段数据"""
        node = Node.query.get(node_id)
        node.updated_by = user_id
        
        node_type = NodeTypeService.get_by_id(node.node_type_id)
        model_name = NodeService.FIELDS_MODELS.get(node_type.slug)
        if model_name:
            fields_model = globals().get(model_name)
            fields_record = fields_model.query.filter_by(node_id=node_id).first()
            if fields_record:
                for key, value in fields_data.items():
                    setattr(fields_record, key, value)
        
        db.session.commit()
        return node
    
    @staticmethod
    def delete(node_id):
        """删除节点：先删字段表记录，再删主表记录"""
        node = Node.query.get(node_id)
        
        # 删除字段记录
        node_type = NodeTypeService.get_by_id(node.node_type_id)
        model_name = NodeService.FIELDS_MODELS.get(node_type.slug)
        if model_name:
            fields_model = globals().get(model_name)
            if fields_model:
                fields_record = fields_model.query.filter_by(node_id=node_id).first()
                if fields_record:
                    db.session.delete(fields_record)
        
        db.session.delete(node)
        db.session.commit()
```

#### 2.3.1 节点类型管理路由 `routes.py`

```python
node_bp = Blueprint('node', __name__, url_prefix='/nodes')

@node_bp.route('/')
def index():
    """节点类型列表"""
    node_types = NodeTypeService.get_all()
    return render_template('node/node_type_list.html', node_types=node_types)

@node_bp.route('/new', methods=['GET', 'POST'])
def create():
    """创建节点类型"""
    form = NodeTypeForm()
    if form.validate_on_submit():
        NodeTypeService.create(form.data)
        flash('节点类型创建成功', 'success')
        return redirect(url_for('node.index'))
    return render_template('node/node_type_edit.html', form=form, action='create')

@node_bp.route('/<int:node_type_id>/edit', methods=['GET', 'POST'])
def edit(node_type_id):
    """编辑节点类型"""
    node_type = NodeTypeService.get_by_id(node_type_id)
    form = NodeTypeForm(obj=node_type)
    if form.validate_on_submit():
        NodeTypeService.update(node_type_id, form.data)
        flash('节点类型更新成功', 'success')
        return redirect(url_for('node.index'))
    return render_template('node/node_type_edit.html', form=form, action='edit', node_type=node_type)

@node_bp.route('/<int:node_type_id>/delete', methods=['POST'])
def delete(node_type_id):
    """删除节点类型"""
    NodeTypeService.delete(node_type_id)
    flash('节点类型已删除', 'success')
    return redirect(url_for('node.index'))

from flask import Blueprint, render_template, redirect, url_for, flash, request
from flask_login import login_required, current_user
from app.services.node.node_type_manager import NodeTypeManager

# 使用 frame_structure 框架
node_bp = Blueprint('node_admin', __name__, url_prefix='/admin/nodes',
                         template_folder='../../templates/node')

@node_bp.route('/')
@login_required
def index():
    """节点类型管理页面 - 显示所有已注册的节点类型"""
    configs = NodeTypeManager.get_all_configs()
    
    # 获取数据库中的节点类型状态
    node_types_status = {}
    for config in configs:
        slug = config.get('slug')
        is_enabled = NodeTypeManager.is_enabled(slug)
        node_types_status[slug] = {
            'enabled': is_enabled,
            'exists': NodeTypeService.get_by_slug(slug) is not None
        }
    
    return render_template('node/node_type_admin.html', 
                         configs=configs,
                         node_types_status=node_types_status)

### 3.4 客户节点服务 `app/services/customer/customer_service.py`

```python
from app.models.node import Node, NodeType
from app.models.node.customer import CustomerFields
from app.services.node.node_type_service import NodeTypeService
from app import db

class CustomerService:
    """客户节点专用服务"""
    
    NODE_TYPE_SLUG = 'customer'
    
    @staticmethod
    def get_node_type():
        return NodeTypeService.get_by_slug(CustomerService.NODE_TYPE_SLUG)
    
    @staticmethod
    def get_customer_list(page=1, per_page=20):
        node_type = CustomerService.get_node_type()
        if not node_type:
            return []
        return Node.query.filter_by(node_type_id=node_type.id).paginate(page=page, per_page=per_page)
    
    @staticmethod
    def get_customer_by_id(customer_id):
        """获取客户节点及字段数据"""
        node = Node.query.get(customer_id)
        if not node:
            return None
        fields = CustomerFields.query.filter_by(node_id=customer_id).first()
        return {'node': node, 'fields': fields}
    
    @staticmethod
    def create_customer(data, user_id):
        """创建客户：先创建主表记录，再创建字段表记录"""
        node_type = CustomerService.get_node_type()
        
        # 创建主表记录
        node = Node(
            node_type_id=node_type.id,
            created_by=user_id,
            updated_by=user_id
        )
        db.session.add(node)
        db.session.flush()
        
        # 创建字段表记录
        fields = CustomerFields(
            node_id=node.id,
            customer_name=data.get('customer_name'),
            contact_person=data.get('contact_person'),
            phone=data.get('phone'),
            email=data.get('email'),
            address=data.get('address'),
            customer_type=data.get('customer_type'),
            notes=data.get('notes')
        )
        db.session.add(fields)
        db.session.commit()
        
        return node
    
    @staticmethod
    def update_customer(customer_id, data, user_id):
        """更新客户字段数据"""
        node = Node.query.get(customer_id)
        node.updated_by = user_id
        
        fields = CustomerFields.query.filter_by(node_id=customer_id).first()
        if fields:
            for key, value in data.items():
                if hasattr(fields, key):
                    setattr(fields, key, value)
        
        db.session.commit()
        return node
    
    @staticmethod
    def delete_customer(customer_id):
        """删除客户：先删字段表记录，再删主表记录"""
        fields = CustomerFields.query.filter_by(node_id=customer_id).first()
        if fields:
            db.session.delete(fields)
        
        node = Node.query.get(customer_id)
        if node:
            db.session.delete(node)
        
        db.session.commit()
    
    @staticmethod
    def get_customer_type_options():
        """获取客户类型下拉选项"""
        from app.services.node.taxonomy_service import TaxonomyService
        taxonomy = TaxonomyService.get_taxonomy_by_slug('customer_type')
        if not taxonomy:
            return []
        items = TaxonomyService.get_items(taxonomy.id)
        return [(item.id, item.name) for item in items]
```

customer_bp = Blueprint('customer', __name__, url_prefix='/nodes/customer')

@customer_bp.route('/')
def index():
    """客户列表"""
    page = request.args.get('page', 1, type=int)
    pagination = CustomerService.get_customer_list(page=page)
    return render_template('node/customer/list.html',

                         pagination=pagination)

@customer_bp.route('/new', methods=['GET', 'POST'])
def create():
    """新建客户"""
    form = CustomerForm()
    form.customer_type.choices = CustomerService.get_customer_type_options()
    
    if form.validate_on_submit():
        data = {
            'customer_name': form.customer_name.data,
            'contact_person': form.contact_person.data,
            'phone': form.phone.data,
            'email': form.email.data,
            'address': form.address.data,
            'customer_type': form.customer_type.data,
            'notes': form.notes.data,
        }
        CustomerService.create_customer(data, current_user.id)
        flash('客户创建成功', 'success')
        return redirect(url_for('customer.index'))
    
    return render_template('node/customer/edit.html', form=form, action='create')

@customer_bp.route('/<int:customer_id>')
def view(customer_id):
    """查看客户"""
    result = CustomerService.get_customer_by_id(customer_id)
    if not result:
        flash('客户不存在', 'danger')
        return redirect(url_for('customer.index'))
    return render_template('node/customer/view.html', node=result['node'], fields=result['fields'])

@customer_bp.route('/<int:customer_id>/edit', methods=['GET', 'POST'])
def edit(customer_id):
    """编辑客户"""
    result = CustomerService.get_customer_by_id(customer_id)
    if not result:
        flash('客户不存在', 'danger')
        return redirect(url_for('customer.index'))
    
    form = CustomerForm(data=result['fields'].__dict__ if result['fields'] else {})
    form.customer_type.choices = CustomerService.get_customer_type_options()
    
    if form.validate_on_submit():
        data = {
            'customer_name': form.customer_name.data,
            'contact_person': form.contact_person.data,
            'phone': form.phone.data,
            'email': form.email.data,
            'address': form.address.data,
            'customer_type': form.customer_type.data,
            'notes': form.notes.data,
        }
        CustomerService.update_customer(customer_id, data, current_user.id)
        flash('客户更新成功', 'success')
        return redirect(url_for('customer.view', customer_id=customer_id))
    
    return render_template('node/customer/edit.html', form=form, action='edit', node=result['node'], fields=result['fields'])

@customer_bp.route('/<int:customer_id>/delete', methods=['POST'])
